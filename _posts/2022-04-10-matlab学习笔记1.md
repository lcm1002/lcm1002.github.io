```matlab
%提取矩阵的子阵
B=magic(5);             %产生一个五阶魔方阵B
B1=B(1:2,[1,3,5]);      %提取B 阵的第1 行，第2 行的第1,3,5 个元素
B2=B([3,1],:);          %提取B 阵的第三行和第一行全部元素
B([1,3],[2,4])=zeros(2);%使得B 阵的第一行和第三行第2，4 个元素为0
L=B(1,:)<5;             %标出B 阵的第一行中小于5 的元素
B3=B(1,L);              %获得B 阵的第一行中小于5 的子向量
```
